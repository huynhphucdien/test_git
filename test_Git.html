B2: git init: 'khởi tạo git ở thư mục làm việc'

B3: git remote add origin 'URL github' 'Liên kết repo local to repo remote'

B4: git add "file" 'thêm file vào repo local'

B5: git commit -m "decribe message" 'Lưu lại những thay đổi vào workspace'

B6: git commit --ammend -m "New message" 'Thay đổi nội dung của commit'

B7: git push origin [branch-name] 'Upload tất cả commit trong local branch lên github'
